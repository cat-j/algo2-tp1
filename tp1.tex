\tadIgualdadObservacional{t}{t'}{tablero}{$(\text{\#}casilleros(t) \igobs \text{\#}casilleros(t'))\ \yluego$\\ $(\forall(c,c':nat)) 1 \leq c,c' \leq \text{\#}casilleros(t) \impluego$\\$ (cont(c,t) \igobs cont(c,t') \land $\\$ movsDesdeHasta(c,c',t) \igobs movsDesdeHasta(c,c',t') ))$}

\tadGeneradores

\begin{flushright}
\textit{//crearTablero: crea un tablero con dos casilleros. El 1 es del primer continente, el 2 del segundo; el primer movimiento va desde el primer casillero hasta el segundo y el segundo movimiento hace lo opuesto.}
\end{flushright}
\tadOperacion{crearTablero}{continente,continente,mov,mov}{tablero}{}

\begin{flushright}
\textit{//agregarCasillero: agrega un casillero del continente k conectado por el movimiento m al casillero c. El movimiento m' conecta c al casillero creado. Las restricciones en k aseguran que se cumpla el agrupamiento de continentes, las restricciones sobre m y m' aseguran que se cumpla la unicidad de movimientos y la necesidad de dos movimientos asegura que se cumpla la simetr\'ia. Si n es la cantidad de casilleros en el momento que esta funci\'on es llamada, el n\'umero del casillero nuevo ser\'a n+1.}
\end{flushright}
\tadOperacion{agregarCasillero}{casillero/c,continente/k,mov/m,mov/m',tablero/t}{tablero}{$1 \leq c \leq \text{\#casilleros($t$)} \yluego (k \igobs cont(c,t) \lor ((\forall c':nat) 1 \leq c' \leq \text{\#casilleros($t$)} \impluego $\\$ cont(c',t) \neq k ) ) \yluego m' \notin \text{todosLosMovs($c,t$)}$}

\begin{flushright}
\textit{//conectar: conecta los casilleros pasados como par\'ametros con el movimiento m del primero al segundo y m' del segundo al primero.}
\end{flushright}
\tadOperacion{conectar}{casillero/c,casillero/c',mov/m,mov/m',tablero/t}{tablero} {$1 \leq c,c' \leq \text{\#casilleros($t$)} \land c \neq c' \yluego m \notin \text{todosLosMovs($c,t$)} \yluego m' \notin \text{todosLosMovs($c',t$)}$}

\begin{flushright}
\textit{//agregarFlecha: agrega un movimiento en un solo sentido. Requiere que los casilleros ya est\'en conectados para que no se rompa la simetr\'ia.}
\end{flushright}
\tadOperacion{agregarFlecha}{casillero/c,casillero/c',mov/m,tablero/t}{tablero} {$1 \leq c, c' \leq \text{\#casilleros($t$)} \yluego \text{conectados?($c,c',t$)} \yluego m \notin \text{todosLosMovs($c,t$)}$}

\textit{//crearPartida: crea una nueva partida cuyo tablero es t. \tadOperacion{crearPartida}{tablero/t,nat/js,secuExt(casillero)/cs,secuExt(continente)/ks}{partida}{$2 \leq js \land \text{long($cs$)} = \text{long($ks$)} = js \land \text{sinRepetidos($cs$)} \land ((\forall k:string) \text{est\'a?($k,ks$)} \Rightarrow $\\$ k \in \text{continentes($t$)}) \land ((\forall c:nat) \text{est\'a?($c,cs$)} \Rightarrow 1 \leq c \leq \text{\#casilleros($t$)})$}

\begin{flushright}

\end{flushright}

\begin{flushright}

\end{flushright}
\tadOperacion{mover}{jugador/j,mov/m,nat/n,partida/p}{partida}{$1 \leq j \leq \text{\#jugadores($p$)} \yluego \text{est\'aActivo?($j,p$)} \land \neg \text{terminada?($p$)}$}
